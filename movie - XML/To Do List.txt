FORMAT: 
NAME - TYPE - OCCURRANCE/REQUIREMENT - DATA TYPE

A  // Attribute
E // Element

R // Required
O // Optional

SC // Simple Custom Data
CC // Complex Custom Data

- Movies - E - 1 - 

- Movie - E -  1+ - 
    - Title - A- R - STRING 
    - Release Date - A - O - CC (sequence integer 1-31 day, integer 1-12 month, integer >=1896 year)
    - Duration - A - O - SC ( integer, >=60 min)
    - Rating - A - O -  CC ( choice - G, PG, PG-13, R, NC-17, X(no longer in use?))
    - Plot - A - O - STRING 
    - Genre - A - R - CC ( see genres.txt )
    - Awards ( nominated, won ) - A - O - STRING?

  - Production
    - Production Company - E - >= 0 - 
      - Company Name - A - R - 
      - Budget - A - R - SC (integer, >=1000$)
      - Shooting Locations - A - O - STRING? ( or sequence CC with Country, City?) 

  - Creative Team - E - 1 - 
    - Director - E - 1+ - 
      - Director's Name - A - R - STRING
      - Biography - A - O - STRING
      - Other Works - A - O - Reference to other movies? 
    - Screenwriter - E - 1+ - 
      - Screenwriter's Name - A - R - STRING
      - Biography - A - O - STRING
      - Other Works - A - O - STRING
    - Composer - E - 1+ -  
      - Composer's Name - A - R - STRING
      - Biography - A - O - STRING
      - Soundtrack - E? 
      - Other Works - A - O - STRING

  - Cast - E - 1 - 
    - Actors - E - 1 -  // List of actors
      - Lead Actors - E - 1 - //List of lead actors
        - Lead Role (Character Name) - E - 1+ - 
          - Actor's Name - A - R - STRING
          - Character Description - A - O - STRING
          - Other Works - A - O - STRING
        - Supporting Role (Character Name) E - 0+ -
          - Actor's Name - A - R - STRING
          - Character Description - A - O - STRING
          - Other Works - A - O - STRING
      - Antagonist (Character Name) - E - 0+ - 
        - Actor's Name - A - R - STRING 
        - Character Description - A - O - STRING
        - Other Works - A - O - STRING
      - Cameos - E - 0+ - 
         - Name - A - R - STRING
  - Post-Production - E - 1 - 
    - Film Editor - E - 1+ - 
      - Editor's Name - A - R - STRING
      - Biography - A - O - STRING
      - Other Works - A - O - STRING
    - Sound Designer - E - 
      - Sound Designer's Name - A - R - STRING
      - Biography  - A - O - STRING
      - Other Works - A - O - STRING
    - Visual Effects (VFX) - E - 1+ -
      - VFX Supervisor's Name - A - R - STRING
      - CGI Details - 
      - Biography  - A - O - STRING
      - Other Works - A - O - STRING
    - Marketing and Publicity - E - 1 - 
      - Marketing Company - E - 1? -
      - Publicity Company - E - 1? - 

  - Release and Distribution - E - 1 -
    - Distribution Company - E - 
      - Distributor's Name
      - Release Strategy
      - Box Office Performance
    - Home Video and Streaming - E - 1
      - Streaming Platforms - E - 0+ - 
      - DVD/Blu-ray Release - E - 0+ - 

  - Audience and Reception - E - 0+ - 
    - Critical Reception - E - 1 - 
      - Reviews - ?
      - Rotten Tomatoes Score - A - R - SC (0.0 <= x <= 10.00)
      - IMDb Rating - A - R - SC (0.0 <= x <= 10.00)
    - Audience Response - E - 1 -
      - Box Office Revenue - A - R - SC ( >= 0 $)
      - Viewer Ratings - A - R - SC (0.0 <= x <= 10.00)
